import q from 'assert';
// 不新增变量
let [l1, m1] = ['akria', 12];
let { ...n1 } = { 'name2': 'akria2', age2: 18 };
q(l1 === 'akria', 'success');
q(m1 === 12, 'success');
q(n1.name2 === 'akria2', 'success');
q(n1.age2 === 18, 'success');
function j1() {
    let [w1, x1] = ['akria3', 13];
    let { ...y1 } = { 'name2': 'akria4', age2: 18 };
    q(w1 === 'akria3', 'success');
    q(x1 === 13, 'success');
    q(y1.name2 === 'akria4', 'success');
    q(y1.age2 === 18, 'success');
}
j1();
// 新增变量
let { newName: o1, ...p1 } = { 'newName': 'akria5', newAge: 18 };
q(o1 === 'akria5', 'success');
q(p1.newAge === 18, 'success');
function k1() {
    return { Propx1: 1, Propy1: 2 };
}
const { Propx1: q1, Propy1: r1 } = k1();
q(q1 === 1, 'success');
q(r1 === 2, 'success');
let s1 = 3;
let t1 = 4;
const { Propx2: u1, Propy2: v1 } = { Propx2: s1, Propy2: t1 };
q(u1 === 3, 'success');
q(v1 === 4, 'success');
