import assert from 'assert';
export function g(o: number, p: number) {
    return o + p;
}
assert.strictEqual(g(1, 2), 3);
function h(m: number, n: number) {
    return m - n;
}
assert.strictEqual(h(3, 1), 2);
namespace i {
    function j(k: string, l: string): string {
        return k + l;
    }
    assert.strictEqual(j("hello", "world"), "helloworld");
}
