import assert from "assert";
// 单实现interface
interface g {
    b1: number;
}
class h implements g {
    b1: number = 2;
}
let i = new h();
assert(i.b1 === 2);
// 多实现interface
interface j {
    c1(): string;
}
interface k<T> {
    d1(): void;
}
class l<U> implements j, k<number> {
    d1(): void {
        let e1 = 'i can swim';
        return;
    }
    c1<d1>(): string {
        return 'i can fly';
    }
    e1: number;
}
let m = new l();
assert(m.c1() === 'i can fly');
// 单实现class
class o {
    f1() { return 4; }
}

    f1(): number {
        return 1;
    }

assert(q.f1() === 1);
// 多实现class
class t {
    g1: number;
    h1(): number {
        return 1;
    }
}
class u {
    i1(c1: string): string {
        return 'hello 4' + c1;
    }
}
class v implements t, u {
    i1(b1: string): string {
        return 'hello 44' + b1;
    }
    g1: number;
    h1(): number {
        return 33;
    }
}

assert(a1.h1() === 33);
assert(a1.i1('55') === 'hello 4455');
