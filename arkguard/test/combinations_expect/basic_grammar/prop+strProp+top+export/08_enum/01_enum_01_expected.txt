import assert from 'assert';
enum g {
    prop1 = 1,
    u = 2
}
assert(g.prop1 === 1);
assert(g.u === 2);
enum h {
    b1 = 1,
    c1,
    left,
    right
}
const direction: h = h.b1;
assert(direction === 1);
assert(h.b1 === 1);
assert(h.c1 === 2);
assert(h.left === 3);
assert(h.right === 4);
assert(h[1] === 'up');
assert(h[2] === 'down');
assert(h[3] === 'left');
assert(h[4] === 'right');
var i;
(function (b1) {
    b1[b1["prop1"] = 1] = "prop1";
    b1[b1["u"] = 2] = "prop2";
})(i || (i = {}));
assert(i.prop1 === 1);
assert(i.u === 2);
assert(i[1] === 'prop1');
assert(i[2] === 'prop2');
let j = 1;
enum k {
    prop1 = 1,
    u = prop1 + j + 1
}
assert(k.prop1 === 1);
assert(k.u === 3);
assert(k[1] === 'prop1');
assert(k[3] === 'prop2');
enum l {
    a = 2,
    b = 3
}
assert(l.a === 2);
assert(l.b === 3);
assert(l[2] === 'a');
assert(l[3] === 'b');
enum m {
    a = (1).valueOf(),
    b = l.a,
    c = l.b.valueOf()
}
assert(m.a === 1);
assert(m.b === 2);
assert(m.c === 3);
assert(m[1] === 'a');
assert(m[2] === 'b');
assert(m[3] === 'c');
module o {
    export namespace v {
        export enum a1 {
            a = 1
        }
    }
}
assert(o.v.a1.a === 1);
assert(o.v.a1[1] === 'a');
module o {
    export namespace v {
        export enum a1 {
            b = o.v.a1.a + 1
        }
    }
}
assert(o.v.a1.b === 2);
assert(o.v.a1[2] === 'b');
export enum p {
    d1 = 1,
    e1 = 2,
    f1 = 4,
    g1 = 5,
    h1 = 6,
    i1 = 0
}
export const q = {
    "j1": p.i1,
    '0': p.d1,
    '1': p.f1,
    '2': p.e1,
    '3': p.g1,
    '4': p.h1,
};
assert(q["j1"] === 0);
assert(q['0'] === 1);
assert(q['1'] === 4);
assert(q['2'] === 2);
assert(q['3'] === 5);
assert(q['4'] === 6);
export enum t {
    l1 = 1 << 1,
    m1 = 1 << 2
}
assert(t.l1 === 2);
assert(t.m1 === 4);
assert(t[2] === 'A');
assert(t[4] === 'B');
