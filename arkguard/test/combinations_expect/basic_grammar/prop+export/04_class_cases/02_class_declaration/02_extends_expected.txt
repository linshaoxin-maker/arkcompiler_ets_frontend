import assert from "assert";
class C1 {
    q: number = 1;
    t(): string {
        return 'c1';
    }
}
class C2 extends C1 {
    u: number = 2;
}
module M1 {
    export class p {
        v: number = 3;
        a1(): number {
            return 31;
        }
        b1(): number {
            return 32;
        }
    }
    class q extends M1.p {
        b1(): number {
            return 42;
        }
    }
    let r = new q();
    assert(r.a1() === 31);
    assert(r.b1() === 42);
}
let insC1 = new C1();
assert(insC1.t() === 'c1');
let insC2 = new C2();
assert(insC2.u === 2);
let insC3 = new M1.p();
assert(insC3.a1() === 31);
assert(insC3.b1() === 32);
