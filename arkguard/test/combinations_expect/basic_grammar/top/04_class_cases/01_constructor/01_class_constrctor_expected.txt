import a from "assert";
// 单个构造器
class b {
    prop1: number;
    constructor(n: number) {
        this.prop1 = n;
    }
    method1() {
    }
}
let c = new b(1);
a(c.prop1 === 1);
// 多个构造器
class d {
    prop2: string;
    constructor(m: 'hello');
    constructor(l: 'bye');
    constructor(k: string);
    constructor(j: any) {
        this.prop2 = j;
    }
    method2() {
    }
}
let e = new d('bee');
a(e.prop2 === 'bee');
// class声明 - 两个构造器不连续
declare namespace f {
    class A3 {
        constructor();
        str3: string;
        constructor(i: number);
    }
}
declare class g {
    constructor();
    str4: string;
    constructor(h: number);
}
